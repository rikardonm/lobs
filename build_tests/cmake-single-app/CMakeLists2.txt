cmake_minimum_required( VERSION 3.22 )

project(pc-central
    VERSION 0.0.1
    LANGUAGES CXX
    DESCRIPTION "PC-build"
)
set (CMAKE_CXX_STANDARD 23)

add_executable(pc-central src/main.cpp)

target_compile_options(
    pc-central
    PRIVATE
    -Wall
    -Wextra
    -Wpedantic
    -Werror
    -Wuninitialized
    -Wno-missing-field-initializers
)
